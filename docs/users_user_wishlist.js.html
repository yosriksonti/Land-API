/**
 * builds a Wishlist Maker
 * @exports buildMakeWishlist 
 * @returns {function}  - Wishlist Maker
 */
export default function buildMakeWishlist () {
    return function makeWishlist ({ id, name, Url, description, price, category} = {}) {

      if (!name || name.length < 3) {
        throw new Error('Wishlist Name must have a value of 3 or more characters.')
      }
      if (!description || description.length < 3) {
        throw new Error('Wishlist Description must have a value of 3 or more characters.')
      }
      if (!Url || Url.length < 3) {
        throw new Error('Wishlist URL must have a value of 3 or more characters.')
      }
      if (!category || category.length < 3) {
        throw new Error('Wishlist Category must have a value of 3 or more characters.')
      }
      if (!price || price < 0 || isNaN(price)) {
        throw new Error('Wishlist Price must have a Valid value.')
      }
      if (!id || id.length < 3) {
        throw new Error('Wishlist Id must have a value of 3 or more characters.')
      }

      return Object.freeze({
        
        getId: () => id,
        getName: () => name,
        getDescription: () => description,
        getUrl: () => Url,
        getCategory: () => category,
        getPrice: () => price

      })
    }
  }
