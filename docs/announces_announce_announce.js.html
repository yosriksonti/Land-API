/**
 * builds a Announce Maker
 * @exports buildMakeAnnounce 
 * @returns {function}  - Announce Maker
 */
export default function buildMakeAnnounce ({ 
}) {
  return function makeAnnounce ({
    id = "",
    titre,
    date_cr = Date.now(),
    date_update = Date.now(),
    description,
    image = "",
    token,
    visib = true,
    action
  } = {}) {
    if(!action){
      throw new Error('Announce Action required.')        
    }
    else {
      switch(action){
        case "postAnnounce" : {
          if (!titre) {
            throw new Error('Announce must have a titre.')
          }
          if (titre.length < 3) {
            throw new Error("Announce's titre  must be longer than 2 characters.")
          }
          if (!description) {
            throw new Error('Announce must have a description.')
          }
          if (description.length < 3) {
            throw new Error("Announce's description titre must be longer than 2 characters.")
          }        
          
         
          return Object.freeze({
            getId: () => id,
            getTitre: () => titre,
            getDate_cr: () => date_cr,
            getDate_update: () => date_update,
            getDescription: () => description,
            getVisib: () => visib,
            getImage: () => image,
          })
        }

      
        case "putAnnounce": {
          if (!id) {
            throw new Error('Announce must have an Id.')
          }
          if (!titre) {
            throw new Error('Announce must have a titre.')
          }
          if (titre.length < 3) {
            throw new Error("Announce's titre  must be longer than 2 characters.")
          }
          if (!description) {
            throw new Error('Announce must have a description.')
          }
          if (description.length < 3) {
            throw new Error("Announce's description titre must be longer than 2 characters.")
          }
          if (!image) {
            throw new Error('Announce must have Image.')
          }
          if (!token || token.length < 50) {
            throw new Error('Announce must have a Valid Token.')
          }
          return Object.freeze({
            getId: () => id,
            getTitre: () => titre,
            getDate_update: () => date_update,
            getDescription: () => description,
            getImage: () => image,
            getToken: () => token
          })
        }

        case "putVisibAnnounce": {
          if (!id) {
            throw new Error('Announce must have an Id.')
          }
          if (visib != true && visib != false ) {
            throw new Error('Announce must have Visibility.')
          }
          if (!token || token.length < 50) {
            throw new Error('Announce must have a Valid Token.')
          }
          return Object.freeze({
            getId: () => id,
            getDate_update: () => date_update,
            getVisib: () => visib,
            getToken: () => token
          })
        }

        case "postLogin": {
          if (!description) {
            throw new Error('Announce must have a description.')
          }
          if (description.length < 3) {
            throw new Error("Announce's description titre must be longer than 2 characters.")
          }
          if (!password) {
            throw new Error('Announce must have a password.')
          }
          if (password.length < 3) {
            throw new Error("Announce's password must be longer than 2 characters.")
          }
          return Object.freeze({
            getPassword: () => password,
            getDescription: () => description
          })
        }

        case "postReset": {
          if (!description) {
            throw new Error('Announce must have a description.')
          }
          if (description.length < 3) {
            throw new Error("Announce's description titre must be longer than 2 characters.")
          }
          return Object.freeze({
            getDescription: () => description
          })
        }

        case "putPassword": {
          if (!id) {
            throw new Error('Announce must have an Id.')
          }
          if (!description) {
            throw new Error('Announce must have a description.')
          }
          if (description.length < 3) {
            throw new Error("Announce's description titre must be longer than 2 characters.")
          }
          if (!password) {
            throw new Error('Announce must have a password.')
          }
          if (password.length < 3) {
            throw new Error("Announce's Password must be longer than 2 characters.")
          }
          if (!newPassword || newPassword.length < 3) {
            throw new Error('Announce must have a Valid New Password.')
          }
          if (!token || token.length < 3) {
            throw new Error('Announce must have a Valid Token.')
          }
          return Object.freeze({
            getId: () => id,
            getDescription: () => description,
            getPassword:() => password,
            getNewPassword:() => newPassword,
            getToken:() => token
          })
          
        }
        default: {
          throw new Error("Invalid Action.");
        }
      }
    }
  }

}
